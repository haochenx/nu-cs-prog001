#include <stdio.h>

int main (void) {
  int a;
  int is_even;
  
  scanf ("%d", &a);
  is_even= a % 2 == 0;

  switch (is_even) {
  case 1:
    // TRUE
    printf("%d is an even number\n", a);
    break;
  case 0:
    // FALSE
    printf("%d is an odd number\n", a);
    break;
  }

  return 0;
}

  AUTHOR   : Haochen Xie

#include <stdio.h>

int main (void) {
  int a, b, c;
  int delta;

  scanf("%d%d%d", &a, &b, &c);
  delta = b * b - 4 * a * c;

  if (delta == 0) {
    printf ("one real root\n");
  } else if (delta > 0) {
    printf ("two distinct real roots\n");
  } else {
    printf ("complex roots\n");
  }

  return 0;
}

#include <stdio.h>

int main (void) {
  int arr[10];
  int i, n, x;

  printf ("Input n, then followed by n integers (n <= 10):\n");
  scanf ("%d", &n);

  for (i=0; i < n; ++i) {
    scanf ("%d", &arr[i]);
  }

  printf ("a    a*a    a*a*a\n");
  for (i=0; i < n; ++i) {
    x = arr[i];
    printf ("%-4d %-6d %-6d\n", x, x * x, x * x * x);
  }

  return 0;
}
